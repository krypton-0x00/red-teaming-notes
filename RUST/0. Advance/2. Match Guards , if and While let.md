```rust
let num = 5;

match num {
    x if x % 2 == 0 => println!("Even number"),
    _ => println!("Odd number"),
}

```

```rust
let p = (3, 7);

match p {
    (x, y) if x == y => println!("Square point"),
    _ => println!("Not a square point"),
}

```

## IF and WHILE LET


```rust
if let Some(val) = Some(5) {
    println!("Value is {}", val);
}

let mut stack = vec![1, 2, 3];
while let Some(top) = stack.pop() {
    println!("Popped: {}", top);
}

```
